with active as (
  select
    day,
    suid,
    dsn
  from `odata-dev.monitoring.daily_sock`
  where monitoring_hours >= 0.25
),

candidates as (
  select
    min(day) as first_active_day,
    suid,
    dsn
  from active
  group by suid, dsn
  having first_active_day >= date("2024-01-01")
),

filtered as (
  select
    c.first_active_day,
    c.suid,
    c.dsn,
    a.day as active_day
  from candidates c
  left join active a
    using (suid, dsn)
),

cohorts as (
  -- assign each subscription to a cohort by the year-month of first activity
  select
    format_date('%y-%m', first_active_day) as cohort_month,
    suid,
    dsn,
    first_active_day,
    active_day,
    -- how many full months after first_active_day is this activity?
    date_diff(
      date_trunc(active_day, month),
      date_trunc(first_active_day, month),
      month
    ) as month_number
  from filtered
),

cohort_sizes as (
  -- count distinct subscriptions per cohort (the denominator)
  select
    cohort_month,
    count(distinct concat(suid, '-', dsn)) as cohort_size
  from cohorts
  where month_number = 0
  group by cohort_month
),

retention_counts as (
  -- count active subscriptions per cohort per month offset
  select
    cohort_month,
    month_number,
    count(distinct concat(suid, '-', dsn)) as active_count
  from cohorts
  where month_number >= 0
  group by cohort_month, month_number
)

select
  r.cohort_month,
  cs.cohort_size as subscriptions,
  r.month_number,
  r.active_count,
  round(r.active_count / cs.cohort_size * 100, 1) as retention_rate
from retention_counts r
join cohort_sizes cs
  using (cohort_month)
order by
  r.cohort_month,
  r.month_number
